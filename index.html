<script>

  // ⚠️ ВСТАВЬ СЮДА СВОЙ URL ИЗ APPS SCRIPT

  const API_URL = 'https://script.google.com/macros/s/AKfycbyXylzVH9IBSQ_A4MzZb1-ZITSN_PpxRam1E6ncXmPymdje1LXwjq1dK1XuOjgKWSx9/exec';


  const tg = window.Telegram.WebApp;

  tg.ready();

  tg.expand();


  let allItems = [];


  async function loadItems() {

    try {

      const res  = await fetch(API_URL + '?action=getAll');

      const data = await res.json();

      allItems   = data.items || [];

      renderList();

      renderExpire();

      updateTabBadges();

    } catch(err) {

      document.getElementById('items-list').innerHTML =

        '<div class="empty">❌ Ошибка загрузки<br>Проверь URL Apps Script</div>';

    }

  }


  function formatDate(raw) {

    if (!raw) return '';

    const d = new Date(raw);

    if (!isNaN(d)) return d.toLocaleDateString('ru-RU');

    return raw;

  }


  function getDaysLeft(item) {

    if (item.daysLeft !== '' && item.daysLeft !== null && !isNaN(item.daysLeft)) {

      return parseInt(item.daysLeft);

    }

    if (!item.expiry) return null;

    const exp = new Date(item.expiry);

    if (isNaN(exp)) return null;

    const now = new Date(); now.setHours(0,0,0,0);

    return Math.ceil((exp - now) / 86400000);

  }


  function expiryBadge(item) {

    const d = getDaysLeft(item);

    if (d === null) return '';

    if (d < 0)   return `<span class="badge badge-danger">❌ Просрочено на ${-d} дн.</span>`;

    if (d === 0) return `<span class="badge badge-danger">⚠️ Истекает СЕГОДНЯ</span>`;

    if (d <= 3)  return `<span class="badge badge-warn">⏳ Осталось ${d} дн.</span>`;

    return       `<span class="badge badge-ok">✅ Годен ещё ${d} дн.</span>`;

  }


  function cardHTML(item) {

    const nameSafe  = item.name.replace(/'/g, "\\'");

    const addedFmt  = formatDate(item.added);

    const expiryFmt = formatDate(item.expiry);

    const doneStyle = item.status === 'done' ? 'opacity:0.5;text-decoration:line-through' : '';

    return `

      <div class="card">

        <div class="card-left">

          <div class="card-name" style="${doneStyle}">${item.name}</div>

          <div class="card-sub">

            Добавлено: ${addedFmt}

            ${expiryFmt ? ' · Годен до: ' + expiryFmt : ''}

          </div>

          ${expiryBadge(item)}

        </div>

        <button class="del-btn" onclick="confirmDelete(${item.rowNum}, '${nameSafe}')">🗑️</button>

      </div>`;

  }


  function renderList() {

    const el = document.getElementById('items-list');

    if (!allItems.length) {

      el.innerHTML = '<div class="empty">🧊 Холодильник пуст<br>Перейди на вкладку ➕ Добавить</div>';

      return;

    }

    const sorted = [...allItems].sort((a, b) => {

      const da = getDaysLeft(a) ?? 9999;

      const db = getDaysLeft(b) ?? 9999;

      return da - db;

    });

    el.innerHTML =

      `<div class="section-title">Позиций в холодильнике: ${allItems.length}</div>` +

      sorted.map(cardHTML).join('');

  }


  function renderExpire() {

    const el  = document.getElementById('expire-list');

    const exp = allItems.filter(i => { const d = getDaysLeft(i); return d !== null && d <= 3; });

    if (!exp.length) {

      el.innerHTML = '<div class="empty">✅ Все продукты свежие!<br><br>Здесь появятся продукты,<br>у которых срок годности<br>меньше 3 дней</div>';

      return;

    }

    el.innerHTML =

      `<div class="section-title">⚠️ Требуют внимания: ${exp.length}</div>` +

      exp.map(cardHTML).join('');

  }


  function updateTabBadges() {

    const expCount = allItems.filter(i => { const d = getDaysLeft(i); return d !== null && d <= 3; }).length;

    document.getElementById('tab-btn-list').innerHTML =

      `📋 Список<span class="counter">${allItems.length}</span>`;

    document.getElementById('tab-btn-expire').innerHTML =

      `⚠️ Истекает` + (expCount ? `<span class="counter">${expCount}</span>` : '');

  }


  async function addItem() {

    const name      = document.getElementById('inp-name').value.trim();

    const expiryRaw = document.getElementById('inp-expiry').value;

    if (!name) { tg.showAlert('⚠️ Введи название продукта'); return; }

    let expiry = '';

    if (expiryRaw) {

      const d = new Date(expiryRaw);

      expiry = d.toLocaleDateString('ru-RU');

    }

    showStatus('Добавляю...');

    try {

      await fetch(API_URL, {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({ action: 'add', name, expiry })

      });

      document.getElementById('inp-name').value   = '';

      document.getElementById('inp-expiry').value = '';

      showStatus('✅ ' + name + ' добавлен!');

      await loadItems();

      showTab('list');

    } catch(err) { tg.showAlert('❌ Ошибка при добавлении'); }

  }


  function confirmDelete(rowNum, name) {

    tg.showConfirm(`Удалить «${name}» из холодильника?`, async (ok) => {

      if (!ok) return;

      showStatus('Удаляю...');

      try {

        await fetch(API_URL, {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({ action: 'delete', rowNum })

        });

        showStatus('🗑️ ' + name + ' удалён');

        await loadItems();

      } catch(err) { tg.showAlert('❌ Ошибка при удалении'); }

    });

  }


  function showTab(tab) {

    ['list', 'add', 'expire'].forEach(t => {

      document.getElementById('tab-' + t).style.display = (t === tab) ? 'block' : 'none';

    });

    document.querySelectorAll('.tab').forEach((el, i) => {

      el.classList.toggle('active', ['list','add','expire'][i] === tab);

    });

  }


  function showStatus(msg) {

    const el = document.getElementById('status');

    el.textContent   = msg;

    el.style.opacity = '1';

    clearTimeout(el._t);

    el._t = setTimeout(() => { el.style.opacity = '0'; }, 2500);

  }


  loadItems();
